   111→function shuffleDeck<T>(deck: T[]): T[] {
   112→  const shuffled = [...deck];
   113→  for (let i = shuffled.length - 1; i > 0; i--) {
   114→    const j = Math.floor(Math.random() * (i + 1));
   115→    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
   116→  }
   117→  return shuffled;
   118→}

   120→function calculateRent(tile: Tile, owner: Player, marketState: MarketState): number {
   121→  if (!tile.rent || tile.rent.length === 0) return 0;
   122→
   123→  // Highway: count owned highways
   124→  if (tile.type === 'highway') {
   125→    const highwayTiles = BOARD_TILES.filter(t => t.type === 'highway');
   126→    const ownedCount = highwayTiles.filter(t => owner.properties.includes(t.id)).length;
   127→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   128→    return Math.round(baseRent * marketState.inflationMultiplier);
   129→  }
   130→
   131→  // Media: count owned media
   132→  if (tile.type === 'media') {
   133→    const mediaTiles = BOARD_TILES.filter(t => t.type === 'media');
   134→    const ownedCount = mediaTiles.filter(t => owner.properties.includes(t.id)).length;
   135→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   136→    return Math.round(baseRent * marketState.inflationMultiplier);
   137→  }
   138→
   139→  // Regular property
   140→  const houses = tile.houses || 0;
   141→  const baseRent = tile.rent[Math.min(houses, tile.rent.length - 1)] || tile.rent[0];
   142→  
   143→  // Check if owner has full color group
   144→  if (tile.colorGroup) {
   145→    const colorGroupTiles = BOARD_TILES.filter(t => t.colorGroup === tile.colorGroup);
   146→    const ownsAll = colorGroupTiles.every(t => owner.properties.includes(t.id));
   147→    if (ownsAll) {
   148→      return Math.round(baseRent * 2 * marketState.inflationMultiplier);
   149→    }
   150→  }
   151→
   152→  // Federal power bonus
   153→  if (tile.colorGroup === 'darkblue') {
   154→    return Math.round(baseRent * marketState.federalRentBonus * marketState.inflationMultiplier);
   155→  }
   156→
   157→  return Math.round(baseRent * marketState.inflationMultiplier);
   158→}
