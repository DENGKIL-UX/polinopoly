    94‚Üí    94‚Üífunction shuffleDeck<T>(deck: T[]): T[] {
    95‚Üí    95‚Üí  const shuffled = [...deck];
    96‚Üí    96‚Üí  for (let i = shuffled.length - 1; i > 0; i--) {
    97‚Üí    97‚Üí    const j = Math.floor(Math.random() * (i + 1));
    98‚Üí    98‚Üí    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
    99‚Üí    99‚Üí  }
   100‚Üí   100‚Üí  return shuffled;
   101‚Üí   101‚Üí}

   103‚Üí   103‚Üífunction calculateRent(tile: Tile, owner: Player, marketState: MarketState): number {
   104‚Üí   104‚Üí  if (!tile.rent || tile.rent.length === 0) return 0;
   105‚Üí   105‚Üí
   106‚Üí   106‚Üí  // Highway: count owned highways
   107‚Üí   107‚Üí  if (tile.type === 'highway') {
   108‚Üí   108‚Üí    const highwayTiles = BOARD_TILES.filter(t => t.type === 'highway');
   109‚Üí   109‚Üí    const ownedCount = highwayTiles.filter(t => owner.properties.includes(t.id)).length;
   110‚Üí   110‚Üí    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   111‚Üí   111‚Üí    return Math.round(baseRent * marketState.inflationMultiplier);
   112‚Üí   112‚Üí  }
   113‚Üí   113‚Üí
   114‚Üí   114‚Üí  // Media: count owned media
   115‚Üí   115‚Üí  if (tile.type === 'media') {
   116‚Üí   116‚Üí    const mediaTiles = BOARD_TILES.filter(t => t.type === 'media');
   117‚Üí   117‚Üí    const ownedCount = mediaTiles.filter(t => owner.properties.includes(t.id)).length;
   118‚Üí   118‚Üí    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   119‚Üí   119‚Üí    return Math.round(baseRent * marketState.inflationMultiplier);
   120‚Üí   120‚Üí  }
   121‚Üí   121‚Üí
   122‚Üí   122‚Üí  // Regular property
   123‚Üí   123‚Üí  const houses = tile.houses || 0;
   124‚Üí   124‚Üí  const baseRent = tile.rent[Math.min(houses, tile.rent.length - 1)] || tile.rent[0];
   125‚Üí   125‚Üí  
   126‚Üí   126‚Üí  // Check if owner has full color group
   127‚Üí   127‚Üí  if (tile.colorGroup) {
   128‚Üí   128‚Üí    const colorGroupTiles = BOARD_TILES.filter(t => t.colorGroup === tile.colorGroup);
   129‚Üí   129‚Üí    const ownsAll = colorGroupTiles.every(t => owner.properties.includes(t.id));
   130‚Üí   130‚Üí    if (ownsAll) {
   131‚Üí   131‚Üí      return Math.round(baseRent * 2 * marketState.inflationMultiplier);
   132‚Üí   132‚Üí    }
   133‚Üí   133‚Üí  }
   134‚Üí   134‚Üí
   135‚Üí   135‚Üí  // Federal power bonus
   136‚Üí   136‚Üí  if (tile.colorGroup === 'darkblue') {
   137‚Üí   137‚Üí    return Math.round(baseRent * marketState.federalRentBonus * marketState.inflationMultiplier);
   138‚Üí   138‚Üí  }
   139‚Üí   139‚Üí
   140‚Üí   140‚Üí  return Math.round(baseRent * marketState.inflationMultiplier);
   141‚Üí   141‚Üí}
