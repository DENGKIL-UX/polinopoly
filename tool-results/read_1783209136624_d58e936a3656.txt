   161→function shuffleDeck<T>(deck: T[]): T[] {
   162→  const shuffled = [...deck];
   163→  for (let i = shuffled.length - 1; i > 0; i--) {
   164→    const j = Math.floor(Math.random() * (i + 1));
   165→    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
   166→  }
   167→  return shuffled;
   168→}

   170→function calculateRent(tile: Tile, owner: Player, marketState: MarketState, mortgagedTiles: number[]): number {
   171→  if (!tile.rent || tile.rent.length === 0) return 0;
   172→
   173→  // Mortgaged properties produce no rent
   174→  if (mortgagedTiles.includes(tile.id)) return 0;
   175→
   176→  // Highway: count owned highways
   177→  if (tile.type === 'highway') {
   178→    const highwayTiles = BOARD_TILES.filter(t => t.type === 'highway');
   179→    const ownedCount = highwayTiles.filter(t => owner.properties.includes(t.id)).length;
   180→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   181→    return Math.round(baseRent * marketState.inflationMultiplier);
   182→  }
   183→
   184→  // Media: count owned media
   185→  if (tile.type === 'media') {
   186→    const mediaTiles = BOARD_TILES.filter(t => t.type === 'media');
   187→    const ownedCount = mediaTiles.filter(t => owner.properties.includes(t.id)).length;
   188→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   189→    return Math.round(baseRent * marketState.inflationMultiplier);
   190→  }
   191→
   192→  // Regular property
   193→  const houses = tile.houses || 0;
   194→  const baseRent = tile.rent[Math.min(houses, tile.rent.length - 1)] || tile.rent[0];
   195→  
   196→  // Check if owner has full color group
   197→  if (tile.colorGroup) {
   198→    const colorGroupTiles = BOARD_TILES.filter(t => t.colorGroup === tile.colorGroup);
   199→    const ownsAll = colorGroupTiles.every(t => owner.properties.includes(t.id));
   200→    if (ownsAll) {
   201→      return Math.round(baseRent * 2 * marketState.inflationMultiplier);
   202→    }
   203→  }
   204→
   205→  // Federal power bonus
   206→  if (tile.colorGroup === 'darkblue') {
   207→    return Math.round(baseRent * marketState.federalRentBonus * marketState.inflationMultiplier);
   208→  }
   209→
   210→  return Math.round(baseRent * marketState.inflationMultiplier);
   211→}
