    94→function shuffleDeck<T>(deck: T[]): T[] {
    95→  const shuffled = [...deck];
    96→  for (let i = shuffled.length - 1; i > 0; i--) {
    97→    const j = Math.floor(Math.random() * (i + 1));
    98→    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
    99→  }
   100→  return shuffled;
   101→}

   103→function calculateRent(tile: Tile, owner: Player, marketState: MarketState): number {
   104→  if (!tile.rent || tile.rent.length === 0) return 0;
   105→
   106→  // Highway: count owned highways
   107→  if (tile.type === 'highway') {
   108→    const highwayTiles = BOARD_TILES.filter(t => t.type === 'highway');
   109→    const ownedCount = highwayTiles.filter(t => owner.properties.includes(t.id)).length;
   110→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   111→    return Math.round(baseRent * marketState.inflationMultiplier);
   112→  }
   113→
   114→  // Media: count owned media
   115→  if (tile.type === 'media') {
   116→    const mediaTiles = BOARD_TILES.filter(t => t.type === 'media');
   117→    const ownedCount = mediaTiles.filter(t => owner.properties.includes(t.id)).length;
   118→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   119→    return Math.round(baseRent * marketState.inflationMultiplier);
   120→  }
   121→
   122→  // Regular property
   123→  const houses = tile.houses || 0;
   124→  const baseRent = tile.rent[Math.min(houses, tile.rent.length - 1)] || tile.rent[0];
   125→  
   126→  // Check if owner has full color group
   127→  if (tile.colorGroup) {
   128→    const colorGroupTiles = BOARD_TILES.filter(t => t.colorGroup === tile.colorGroup);
   129→    const ownsAll = colorGroupTiles.every(t => owner.properties.includes(t.id));
   130→    if (ownsAll) {
   131→      return Math.round(baseRent * 2 * marketState.inflationMultiplier);
   132→    }
   133→  }
   134→
   135→  // Federal power bonus
   136→  if (tile.colorGroup === 'darkblue') {
   137→    return Math.round(baseRent * marketState.federalRentBonus * marketState.inflationMultiplier);
   138→  }
   139→
   140→  return Math.round(baseRent * marketState.inflationMultiplier);
   141→}

   222→  rollDice: () => {
   223→    const d1 = Math.floor(Math.random() * 6) + 1;
   224→    const d2 = Math.floor(Math.random() * 6) + 1;
   225→    const doubles = d1 === d2;
   226→    const state = get();

   237→    if (newConsecutive >= 3) {
   238→      const cp = state.players.find(p => p.id === state.turnOrder[state.currentTurnIndex]);
   239→      if (cp) {
   240→        set(state => ({
   241→          players: state.players.map(p =>
   242→            p.id === cp.id ? { ...p, position: 10, isInJail: true, jailTurns: 0 } : p

   251→        });
   252→      }
   253→      return;

   296→  handleLanding: () => {

   299→    const player = state.players.find(p => p.id === currentPlayerId);

   307→        players: state.players.map(p =>
   308→          p.id === currentPlayerId ? { ...p, position: 10, isInJail: true, jailTurns: 0 } : p
   309→        ),
   310→        phase: 'jail_decision',

   313→        playerId: currentPlayerId,
   314→        playerName: player.name,

   316→        type: 'jail',
   317→      });
   318→      return;
   319→    }
   320→
   321→    // Tax tiles
   322→    if (tile.type === 'tax') {
   323→      const amount = tile.id === 4 ? 200 : 100;
   324→      const newMoney = player.money - amount;
   325→      const isBankrupt = newMoney < 0;
   326→      set(state => ({
   327→        players: state.players.map(p =>
   328→          p.id === currentPlayerId ? { ...p, money: newMoney, isBankrupt } : p
   329→        ),
   330→        phase: isBankrupt ? 'game_over' : 'landed',
   331→      }));
   332→      get().addLog({
   333→        playerId: currentPlayerId,
   334→        playerName: player.name,
   335→        message: `💸 ${player.name} pays RM${amount} in ${tile.name}! (${tile.description})`,
   336→        type: 'tax',
   337→      });
   338→      return;
   339→    }
   340→
   341→    // Card tiles
   342→    if (tile.type === 'chest' || tile.type === 'chance') {
   343→      const card = get().drawCard(tile.type === 'chest' ? 'chest' : 'chance');
   344→      set({ currentCard: card, phase: 'card' });
   345→      get().addLog({
   346→        playerId: currentPlayerId,
   347→        playerName: player.name,
   348→        message: `🃏 ${player.name} draws: "${card.title}"`,
   349→        type: 'card',
   350→      });
   351→      return;
   352→    }
   353→
   354→    // Property / Highway / Media
   355→    if (tile.type === 'property' || tile.type === 'highway' || tile.type === 'media') {
   356→      // Unowned
   357→      if (!tile.owner) {
   358→        if (player.money >= (tile.price || 0)) {
   359→          if (player.isAI) {
   360→            // AI decides — try LLM API, fallback to random
   361→            set({ aiThinking: true });
   362→            const aiDecide = async () => {
   363→              try {
   364→                const res = await fetch('/api/ai-decision', {
   365→                  method: 'POST',
   366→                  headers: { 'Content-Type': 'application/json' },
   367→                  body: JSON.stringify({
   368→                    coalitionId: player.coalitionId,
   369→                    coalitionName: COALITIONS[player.coalitionId]?.fullName || player.name,
   370→                    property: { name: tile.name, id: tile.id, price: tile.price, rent: tile.rent },
   371→                    money: player.money,
   372→                    ownedProperties: player.properties.map(pid => BOARD_TILES[pid]?.name || `Tile ${pid}`),
   373→                    marketData: get().marketState,
   374→                  }),
   375→                });
   376→                const data = await res.json();
   377→                const quote = data.quote || getRandomQuote(player.coalitionId);
   378→                if (data.action === 'BUY' && player.money >= (tile.price || 0)) {
   379→                  get().buyProperty();
   380→                } else {
   381→                  get().skipBuy();
   382→                }
   383→                set(state => ({
   384→                  players: state.players.map(p =>
   385→                    p.id === currentPlayerId ? { ...p, quote } : p
   386→                  ),
   387→                  aiThinking: false,
   388→                }));
   389→                get().addLog({
   390→                  playerId: currentPlayerId,
   391→                  playerName: player.name,
   392→                  message: `🤖 ${player.name}: "${quote}"`,
   393→                  type: 'ai_quote',
   394→                });
   395→              } catch {
   396→                // Fallback to random decision
   397→                const shouldBuy = Math.random() > 0.2;
   398→                const quote = getRandomQuote(player.coalitionId);
   399→                if (shouldBuy && player.money >= (tile.price || 0)) {
   400→                  get().buyProperty();
   401→                } else {
   402→                  get().skipBuy();
   403→                }
   404→                set(state => ({
   405→                  players: state.players.map(p =>
   406→                    p.id === currentPlayerId ? { ...p, quote } : p
   407→                  ),
   408→                  aiThinking: false,
   409→                }));
   410→                get().addLog({
   411→                  playerId: currentPlayerId,
   412→                  playerName: player.name,
   413→                  message: `🤖 ${player.name}: "${quote}"`,
   414→                  type: 'ai_quote',
   415→                });
   416→              }

   426→            message: `😅 ${player.name} cannot afford ${tile.name} (RM${tile.price})!`,

   557→  applyCard: (card: GameCard) => {

   559→    const currentPlayerId = state.turnOrder[state.currentTurnIndex];
   560→    const player = state.players.find(p => p.id === currentPlayerId);
   561→    if (!player) return;
   562→
   563→    const { effect } = card;
   564→
   565→    switch (effect.type) {
   566→      case 'money': {
   567→        const newMoney = player.money + effect.value;
   568→        set(state => ({
   569→          players: state.players.map(p =>
   570→            p.id === currentPlayerId ? { ...p, money: newMoney, isBankrupt: newMoney < 0 } : p
   571→          ),
   572→          currentCard: null,
   573→          phase: newMoney < 0 ? 'game_over' : 'landed',
   574→        }));
   575→        break;
   576→      }
   577→      case 'move': {
   578→        let newPos = player.position + effect.value;
   579→        if (newPos < 0) newPos += 40;
   580→        if (newPos >= 40) {
   581→          newPos = newPos % 40;
   582→          // Pass GO
   583→          set(state => ({
   584→            players: state.players.map(p =>
   585→              p.id === currentPlayerId ? { ...p, money: p.money + 200, position: newPos } : p
   586→            ),
   587→            currentCard: null,
   588→            phase: 'landed',
   589→          }));
   590→          get().addLog({
   591→            playerId: currentPlayerId,
   592→            playerName: player.name,
   593→            message: `💰 Passed GO during card! +RM200!`,
   594→            type: 'pass_go',
   595→          });
   596→        } else {
   597→          set(state => ({
   598→            players: state.players.map(p =>
   599→              p.id === currentPlayerId ? { ...p, position: newPos } : p
   600→            ),
   601→            currentCard: null,
   602→            phase: 'landed',
   603→          }));
   604→        }
   605→        break;
   606→      }
   607→      case 'go_to': {
   608→        const targetPos = effect.tileId || 0;
   609→        const passedGo = player.position > targetPos;
   610→        set(state => ({
   611→          players: state.players.map(p =>
   612→            p.id === currentPlayerId
   613→              ? { ...p, position: targetPos, money: passedGo ? p.money + 200 : p.money }
   614→              : p
   615→          ),
   616→          currentCard: null,
   617→          phase: 'landed',
   618→        }));
   619→        if (passedGo) {
   620→          get().addLog({
   621→            playerId: currentPlayerId,
   622→            playerName: player.name,
   623→            message: `💰 Passed GO during card! +RM200!`,
   624→            type: 'pass_go',
   625→          });
   626→        }
   627→        break;
   628→      }
   629→      case 'jail': {
   630→        set(state => ({
   631→          players: state.players.map(p =>
   632→            p.id === currentPlayerId ? { ...p, position: 10, isInJail: true, jailTurns: 0 } : p
   633→          ),
   634→          currentCard: null,
   635→          phase: 'jail_decision',
   636→        }));
   637→        break;
   638→      }
   639→      case 'collect_all': {
   640→        const amount = effect.value;
   641→        const totalCollected = (state.players.length - 1) * amount;
   642→        set(state => ({
   643→          players: state.players.map(p => {
   644→            if (p.id === currentPlayerId) return { ...p, money: p.money + totalCollected };
   645→            return { ...p, money: p.money - amount };
   646→          }),
   647→          currentCard: null,
   648→          phase: 'landed',
   649→        }));
   650→        break;
   651→      }
   652→      case 'pay_all': {
   653→        const amount = effect.value;
   654→        const totalPaid = (state.players.length - 1) * amount;
   655→        set(state => ({
   656→          players: state.players.map(p => {
   657→            if (p.id === currentPlayerId) return { ...p, money: p.money - totalPaid };
   658→            return { ...p, money: p.money + amount };
   659→          }),
   660→          currentCard: null,
   661→          phase: 'landed',
   662→        }));
   663→        break;
   664→      }
   665→      case 'repair': {
   666→        const cost = effect.value;
   667→        set(state => ({
   668→          players: state.players.map(p =>
   669→            p.id === currentPlayerId ? { ...p, money: p.money - cost } : p
   670→          ),
   671→          currentCard: null,
   672→          phase: 'landed',
   673→        }));
   674→        break;
   675→      }
   676→    }
   677→
   678→    // Check for bankruptcy
   679→    const newState = get();
   680→    const updatedPlayer = newState.players.find(p => p.id === currentPlayerId);
   681→    if (updatedPlayer && updatedPlayer.money < 0) {
   682→      set({ phase: 'game_over' });
   683→    }
   684→  },
   685→
   686→  endTurn: () => {
   687→    const state = get();
   688→    
   689→    // Check game over
   690→    const activePlayers = state.players.filter(p => !p.isBankrupt);
   691→    if (activePlayers.length <= 1) {
   692→      const winner = activePlayers[0]?.id || null;
   693→      set({ phase: 'game_over', winner });
   694→      return;
   695→    }
   696→
   697→    // Next player
   698→    let nextIndex = (state.currentTurnIndex + 1) % state.turnOrder.length;
   699→    let nextPlayerId = state.turnOrder[nextIndex];
   700→    let nextPlayer = state.players.find(p => p.id === nextPlayerId);
   701→
   702→    // Skip bankrupt players
   703→    let safety = 0;
   704→    while (nextPlayer?.isBankrupt && safety < 10) {
   705→      nextIndex = (nextIndex + 1) % state.turnOrder.length;
   706→      nextPlayerId = state.turnOrder[nextIndex];
   707→      nextPlayer = state.players.find(p => p.id === nextPlayerId);
   708→      safety++;
   709→    }
   710→
   711→    const isNextTurn = nextIndex <= state.currentTurnIndex;
   712→    const newTurnCount = isNextTurn ? state.turnCount + 1 : state.turnCount;
   713→
   714→    set({
   715→      currentTurnIndex: nextIndex,
   716→      turnCount: newTurnCount,
   717→      diceValues: null,
   718→      isDoubles: false,
   719→      consecutiveDoubles: 0,
   720→      phase: 'playing',
   721→      selectedTileId: null,
   722→    });
   723→
   724→    // If next player is AI, auto-play
   725→    if (nextPlayer?.isAI) {
   726→      get().setAIThinking(true);
   727→      setTimeout(() => {
   728→        get().aiTurn();
   729→      }, 1000);
   730→    }
   731→  },

   733→  handleJailDecision: (pay: boolean) => {
   734→    const state = get();
   735→    const currentPlayerId = state.turnOrder[state.currentTurnIndex];
   736→    const player = state.players.find(p => p.id === currentPlayerId);
   737→    if (!player) return;
   738→
   739→    if (pay || player.hasGetOutOfJailFree) {
   740→      const cost = pay ? 50 : 0;
   741→      set(state => ({
   742→        players: state.players.map(p =>
   743→          p.id === currentPlayerId
   744→            ? { ...p, money: p.money - cost, isInJail: false, jailTurns: 0, hasGetOutOfJailFree: pay ? p.hasGetOutOfJailFree : false }
   745→            : p
   746→        ),
   747→        phase: 'playing',
   748→      }));
   749→      get().addLog({
   750→        playerId: currentPlayerId,
   751→        playerName: player.name,
   752→        message: `🔓 ${player.name} ${pay ? `pays RM50 bail` : 'uses Get Out of Jail Free card'}!`,
   753→        type: 'jail',
   754→      });
   755→    } else {
   756→      // Try to roll doubles
   757→      const d1 = Math.floor(Math.random() * 6) + 1;
   758→      const d2 = Math.floor(Math.random() * 6) + 1;
   759→      if (d1 === d2) {
   760→        set(state => ({
   761→          players: state.players.map(p =>
   762→            p.id === currentPlayerId ? { ...p, isInJail: false, jailTurns: 0 } : p
   763→          ),
   764→          diceValues: [d1, d2],
   765→          isDoubles: true,
   766→          phase: 'rolling',
   767→        }));
   768→        get().addLog({
   769→          playerId: currentPlayerId,
   770→          playerName: player.name,
   771→          message: `🎲 ${player.name} rolls doubles (${d1},${d2}) and escapes jail!`,
   772→          type: 'jail',
   773→        });
   774→        setTimeout(() => get().movePlayer(), 1500);
   775→      } else {
   776→        const newJailTurns = player.jailTurns + 1;
   777→        const forcedOut = newJailTurns >= 3;
   778→        set(state => ({
   779→          players: state.players.map(p =>
   780→            p.id === currentPlayerId
   781→              ? {
   782→                  ...p,
   783→                  jailTurns: newJailTurns,
   784→                  isInJail: !forcedOut,
   785→                  money: forcedOut ? p.money - 50 : p.money,
   786→                }
   787→              : p
   788→          ),
   789→          diceValues: [d1, d2],
   790→          isDoubles: false,
   791→          phase: forcedOut ? 'landed' : 'jail_decision',
   792→        }));
   793→        get().addLog({
   794→          playerId: currentPlayerId,
   795→          playerName: player.name,
   796→          message: forcedOut
   797→            ? `⏰ ${player.name} serves full sentence! Pays RM50 and is released.`
   798→            : `Dice: ${d1},${d2} — No doubles. ${player.name} remains in Tahanan SPR. (Turn ${newJailTurns}/3)`,
   799→          type: 'jail',
   800→        });
   801→      }
   802→    }
   803→  },

   823→  aiTurn: async () => {
   824→    const state = get();
   825→    const currentPlayerId = state.turnOrder[state.currentTurnIndex];
   826→    const player = state.players.find(p => p.id === currentPlayerId);
   827→    if (!player || !player.isAI || player.isBankrupt) {
   828→      get().endTurn();
   829→      return;
   830→    }
   831→
   832→    // Handle jail for AI
   833→    if (player.isInJail) {
   834→      get().handleJailDecision(Math.random() > 0.5);
   835→      await new Promise(resolve => setTimeout(resolve, 1500));
   836→    }
   837→
   838→    const currentState = get();
   839→    if (currentState.phase === 'jail_decision') {
   840→      // Still in jail, end turn
   841→      get().endTurn();
   842→      return;
   843→    }
   844→
   845→    // Roll dice
   846→    get().rollDice();
   847→
   848→    // Wait for move to complete
   849→    await new Promise(resolve => setTimeout(resolve, 3000));
   850→
   851→    // Check if we need to pay rent or buy
   852→    const afterMoveState = get();
   853→    if (afterMoveState.phase === 'paying_rent') {
   854→      get().payRent();
   855→      await new Promise(resolve => setTimeout(resolve, 500));
   856→    }
   857→    if (afterMoveState.phase === 'card') {
   858→      const card = afterMoveState.currentCard;
   859→      if (card) {
   860→        get().applyCard(card);
   861→        await new Promise(resolve => setTimeout(resolve, 500));
   862→      }
   863→    }
   864→
   865→    get().setAIThinking(false);
   866→    get().endTurn();
   867→  },
