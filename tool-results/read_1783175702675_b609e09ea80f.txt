   111→   111→function shuffleDeck<T>(deck: T[]): T[] {
   112→   112→  const shuffled = [...deck];
   113→   113→  for (let i = shuffled.length - 1; i > 0; i--) {
   114→   114→    const j = Math.floor(Math.random() * (i + 1));
   115→   115→    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
   116→   116→  }
   117→   117→  return shuffled;
   118→   118→}

   120→   120→function calculateRent(tile: Tile, owner: Player, marketState: MarketState): number {
   121→   121→  if (!tile.rent || tile.rent.length === 0) return 0;
   122→   122→
   123→   123→  // Highway: count owned highways
   124→   124→  if (tile.type === 'highway') {
   125→   125→    const highwayTiles = BOARD_TILES.filter(t => t.type === 'highway');
   126→   126→    const ownedCount = highwayTiles.filter(t => owner.properties.includes(t.id)).length;
   127→   127→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   128→   128→    return Math.round(baseRent * marketState.inflationMultiplier);
   129→   129→  }
   130→   130→
   131→   131→  // Media: count owned media
   132→   132→  if (tile.type === 'media') {
   133→   133→    const mediaTiles = BOARD_TILES.filter(t => t.type === 'media');
   134→   134→    const ownedCount = mediaTiles.filter(t => owner.properties.includes(t.id)).length;
   135→   135→    const baseRent = tile.rent[Math.min(ownedCount - 1, tile.rent.length - 1)] || tile.rent[0];
   136→   136→    return Math.round(baseRent * marketState.inflationMultiplier);
   137→   137→  }
   138→   138→
   139→   139→  // Regular property
   140→   140→  const houses = tile.houses || 0;
   141→   141→  const baseRent = tile.rent[Math.min(houses, tile.rent.length - 1)] || tile.rent[0];
   142→   142→  
   143→   143→  // Check if owner has full color group
   144→   144→  if (tile.colorGroup) {
   145→   145→    const colorGroupTiles = BOARD_TILES.filter(t => t.colorGroup === tile.colorGroup);
   146→   146→    const ownsAll = colorGroupTiles.every(t => owner.properties.includes(t.id));
   147→   147→    if (ownsAll) {
   148→   148→      return Math.round(baseRent * 2 * marketState.inflationMultiplier);
   149→   149→    }
   150→   150→  }
   151→   151→
   152→   152→  // Federal power bonus
   153→   153→  if (tile.colorGroup === 'darkblue') {
   154→   154→    return Math.round(baseRent * marketState.federalRentBonus * marketState.inflationMultiplier);
   155→   155→  }
   156→   156→
   157→   157→  return Math.round(baseRent * marketState.inflationMultiplier);
   158→   158→}
